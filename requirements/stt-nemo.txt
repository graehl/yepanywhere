# Optional NeMo ASR prototype dependencies for the pixi `stt` environment.
#
# Install from the YA checkout after the base Whisper/Transformers Parakeet
# bootstrap:
#
#   pixi run --frozen -e stt python -m pip install --requirement requirements/stt-nemo.txt
#
# Keep this constrained to the known-good STT snapshot so NeMo cannot silently
# downgrade or replace the working faster-whisper / Transformers Parakeet stack.
-c requirements/stt-known-good-2026-06-16.txt

# 2.7.x conflicts with the current fsspec pin and wants to replace the
# Transformers git build. 2.0.0 coexists with the current pins and can load the
# NeMo 1.1B Parakeet RNNT/CTC models with the runtime NumPy shim documented in
# topics/pluggable-speech-recognition.md.
nemo_toolkit[asr]==2.0.0

# The resolver otherwise selects a newer pytorch-lightning that no longer
# exports NeptuneLogger, which NeMo 2.0.0 imports at ASR module load time.
pytorch-lightning==2.4.0

# NeMo ASR imports VAD utilities that import matplotlib even for inference.
matplotlib>=3.11,<4
